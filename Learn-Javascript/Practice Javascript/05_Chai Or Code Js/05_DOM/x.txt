function findAllSameIdInnerText() {
  const elements = document.querySelectorAll("#video-title"); // Select all elements

  const duplicates = {}; // Object to store duplicate IDs and their innerText

  elements.forEach(element => {
    const id = element.id;
    const innerText = element.innerText.trim(); // Trim whitespace

    if (id && innerText) { // Check if both id and innerText exist
      if (!duplicates[id]) {
        duplicates[id] = []; // Array to store innerText for each unique ID
      }
      duplicates[id].push(innerText); // Add innerText to the array for its ID
    }
  });

  // Print results (modify as needed)
  for (const id in duplicates) {
    if (duplicates[id].length > 1) {
      console.log(`Element with id "${id}" has these inner texts:`);
      duplicates[id].forEach(text => console.log(`  - "${text}"`));
    }
  }
}
